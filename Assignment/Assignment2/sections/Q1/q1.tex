	&= \lambda^2 + \lambda\\

	&= \left.\frac{d}{dz}z(\frac{d}{dz}\lambda e^{(z-1)\lambda})\\
	&= \lambda^3 + 3\lambda^2 + \lambda\\
	\text{Third Central Moment: } \mu_3 &= E[(X-E[X])^3]\\
