\subsection{Part 3}

\begin{align*}
	\pi^{(n)} &= \pi^{(n-1)}P \\
	\text{Given: } n &= 3 \\
	\pi^{0} &= 1 \\
	\pi^{(1)} &= \pi^{(0)}P  = 1(0.5) \\
		  &= 0.5 \\
	\pi^{(2)} &= \pi^{(1)}P = 0.5(0.5) \\
		&= 0.25 \\
	\pi^{(3)} &= \pi^{(2)}P = 0.25(0.5) \\
		  &= 0.125 \\
	\therefore P(X=4 | n=3 ) &= 1-0.125 \\
				 &= \frac{7}{8}
.\end{align*}
